9/**
10 * @brief Nash-Panfilov active stress model.
11 *
12 * This class implements the Nash-Panfilov active stress model [1], with the
13 * modifications introduced by Goktepe and Kuhl [2].
14 *
15 * The model equations are the following:
16 * @f[ \begin{aligned}
17 * \dv{\Tact}{t} &= \varepsilon(\calcium)(
18 * \eta_\text{T} (\calcium - \calcium_\text{rest}) - \Tact)\;, \\
19 * \varepsilon(\calcium) &=
20 * \varepsilon_0 + (\varepsilon_i - \varepsilon_0)
21 * \exp(-\exp(-\xi_T (\calcium - \calcium_\text{crit})))\;,
22 * \end{aligned} @f]
23 * where @f$\eta_\text{T}@f$, @f$\calcium_\text{rest}@f$,
24 * @f$\calcium_\text{crit}@f$, @f$\varepsilon_0@f$, @f$\varepsilon_i@f$ and
25 * @f$\xi_T@f$ are user-defined model parameters. The function
26 * @f$\varepsilon(\calcium)@f$ is a sigmoidal-shaped calcium-dependent time
27 * constant (see Figure 3 in [2] for more details).
28 *
29 * @note The sensitivity of the model to calcium is controlled by the paramter
30 * @f$\eta_\text{T}@f$, which has the same units of active tension over calcium.
31 * Therefore, if the ionic model providing the calcium is phenomenological (see
32 * @ref IonicModel) and calcium is non-dimensional, this parameter may need to
33 * be rescaled as well. Similar considerations apply to @f$\xi_T@f$.
34 *
35 * **References**:
36 * 1. [Nash, Panfilov (2004)](https://doi.org/10.1016/j.pbiomolbio.2004.01.016)
37 * 2. [Goktepe, Kuhl (2009)](https://doi.org/10.1007/s00466-009-0434-z)
38 */

54 add_parameter("calcium_crit", 1.0, required);

56 }
57 };
